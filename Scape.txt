-- Dragging functionality (works on mobile too)
local dragging = false
local dragInput
local dragStart
local startPos

local function updateDrag(input)
    local delta = input.Position - dragStart
    mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

titleBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

titleBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        updateDrag(input)
    end
end)

-- Slider functionality
local sliderActive = false

local function updateSlider(input)
    local pos = input.Position.X
    local bgPos = sliderBg.AbsolutePosition.X
    local bgSize = sliderBg.AbsoluteSize.X
    
    local percent = math.clamp((pos - bgPos) / bgSize, 0, 1)
    sliderFill.Size = UDim2.new(percent, 0, 1, 0)
    
    local delayMs = math.floor(percent * 200) -- 0-200ms range
    delayLabel.Text = "Delay: " .. delayMs .. "ms"
end

sliderFill.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        sliderActive = true
        updateSlider(input)
    end
end)

sliderBg.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        sliderActive = true
        updateSlider(input)
    end
end)

sliderKnob.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        sliderActive = true
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if sliderActive and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        updateSlider(input)
    end
end)

    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        sliderActive = false
    end
end)

-- Find nearest plot function
local function findNearestPlot()
    local plots = Workspace:FindFirstChild("Plots")
    if not plots then return nil end
    
    local character = Players.LocalPlayer.Character
    if not character then return nil end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    
    local nearestPlot = nil
    local nearestDistance = math.huge
    
    for _, plot in pairs(plots:GetChildren()) do
        if plot:IsA("Model") or plot:IsA("Part") then
            local plotPos = plot:IsA("Model") and plot.PrimaryPart and plot.PrimaryPart.Position or plot.Position
            local distance = (hrp.Position - plotPos).Magnitude
            
            if distance < nearestDistance then
                nearestDistance = distance
                nearestPlot = plot
            end
        end
    end
    
    return nearestPlot, nearestDistance
end

-- TP + Grab button functionality
grabButton.MouseButton1Click:Connect(function()
    -- Visual feedback
    grabButton.BackgroundColor3 = Color3.fromRGB(200, 150, 0)
    grabButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    
    local character = Players.LocalPlayer.Character
    if not character then 
        print("No character found")
        -- Reset button
        task.wait(0.1)
        grabButton.BackgroundColor3 = Color3.fromRGB(255, 200, 0)
        grabButton.TextColor3 = Color3.fromRGB(0, 0, 0)
        return 
    end
    
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not hrp then 
        print("No HumanoidRootPart found")
        -- Reset button
        task.wait(0.1)
        grabButton.BackgroundColor3 = Color3.fromRGB(255, 200, 0)
        grabButton.TextColor3 = Color3.fromRGB(0, 0, 0)
        return 
    end
    
    -- Find nearest plot
    local nearestPlot, distance = findNearestPlot()
    if not nearestPlot then
        print("No plots found")
        -- Reset button
        task.wait(0.1)
        grabButton.BackgroundColor3 = Color3.fromRGB(255, 200, 0)
        grabButton.TextColor3 = Color3.fromRGB(0, 0, 0)
        return
    end
    
    print("Found plot: " .. nearestPlot.Name .. " at distance: " .. math.floor(distance))
    
    -- Get teleport position (center of plot or slightly above)
    local tpPosition
    if nearestPlot:IsA("Model") and nearestPlot.PrimaryPart then
        tpPosition = nearestPlot.PrimaryPart.Position + Vector3.new(0, 5, 0)
    elseif nearestPlot:IsA("Part") then
        tpPosition = nearestPlot.Position + Vector3.new(0, 5, 0)
    else
        -- Try to find a part in the model
        local parts = nearestPlot:GetDescendants()
        for _, part in pairs(parts) do
            if part:IsA("Part") or part:IsA("MeshPart") then
                tpPosition = part.Position + Vector3.new(0, 5, 0)
                break
            end
        end
    end
    
    if not tpPosition then
        print("Could not determine teleport position")
        -- Reset button
        task.wait(0.1)
        grabButton.BackgroundColor3 = Color3.fromRGB(255, 200, 0)
        grabButton.TextColor3 = Color3.fromRGB(0, 0, 0)
        return
    end
    
    -- Get current delay from slider
    local percent = sliderFill.Size.X.Scale
    local delayMs = percent * 200
    
    -- Find tool
    local tool = character:FindFirstChildWhichIsA("Tool")
    if tool then
        print('Using tool: ' .. tool.Name)
        
        -- Teleport to plot
        hrp.CFrame = CFrame.new(tpPosition)
        print("Teleported to plot")
        
        -- Wait for delay
        if delayMs > 0 then
            task.wait(delayMs / 1000)
        end
        
        -- Activate tool (grab)
        tool:Activate()
        print("Tool activated - GRAB executed")
    else
        print("No tool found in character")
        -- Just teleport if no tool
        hrp.CFrame = CFrame.new(tpPosition)
        print("Teleported to plot (no tool)")
    end
    
    -- Reset button
    task.wait(0.1)
    grabButton.BackgroundColor3 = Color3.fromRGB(255, 200, 0)
    grabButton.TextColor3 = Color3.fromRGB(0, 0, 0)
end)

-- Update ping and FPS
local lastTime = tick()
local lastPing = 0

RunService.RenderStepped:Connect(function()
    -- Update ping
    local ping = Players.LocalPlayer:GetNetworkPing() * 1000
    pingLabel.Text = "Ping: " .. math.floor(ping) .. "ms"
    
    -- Update FPS
    local currentTime = tick()
    local fps = math.floor(1 / (currentTime - lastTime))
    lastTime = currentTime
    
    statsLabel.Text = fps .. " | " .. math.floor(ping)
    
    -- Update stats label color based on ping
    if ping < 100 then
        statsLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
    elseif ping < 200 then
        statsLabel.TextColor3 = Color3.fromRGB(255, 200, 0)
    else
        statsLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
    end
end)
